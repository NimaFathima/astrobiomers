```typescript

import { tourSteps } from '@/data/tourScript';

function Home() {
  const { showTour, startTour, endTour } = useTour();

  return (
    <div>
      {/* Start Tour Button */}
      <button
        onClick={startTour}
        className="px-6 py-3 bg-purple-600 text-white rounded"
      >
        üé§ Take the Tour
      </button>

      {/* Tour Component */}
      {showTour && (
        <GuidedTour
          steps={tourSteps}
          onComplete={endTour}
          onSkip={endTour}
          autoStart={true}
        />
      )}
    </div>
  );
}
```
